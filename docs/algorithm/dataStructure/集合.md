# 集合 Collection


- 概念
  - 空集
  - 子集
  - 交集
  - 并集
  - 差集
- 特征： 无重复性

Set 强引用
WeakSet 弱引用

## 场景

## 方法

- 添加值 add(value)
- 检查值 has(value)
- 删除值 delete(value)
- 清除集合 clear()
- 获取集合长度 size()
- 并集 union()
- 交集 intersection()
- 差集 difference()

## 实现

```js

class Set {
    constructor() {
        this.items = {}
    }
    // 检查
    has(value) {
        return this.items.hasOwnProperty(value);
    }
    // 添加
    add(value) {
        if(!this.has(value)) {
            this.items[value] = value;
            return value;
        }
        return false;
    }
    // 删除
    delete(value) {
        if(this.has(value)) {
            delete this.items[value]
            return true
        }
        return null
    }
    // 清除集合
    clear() {
        this.items = {}
    }
    // 获取集合长度
    size() {
        return Object.keys(this.items).length
    }

    value() {
        let values = []
        for(let k in this.items) {
            if(this.items.hasOwnProperty(k)) {
                values.push(this.items[key])
            }
        }
        return values
    }
    // 并集
    union(otherSet) {
        let resultSet = new Set()

        // 1.取出自己的值
        let arr = this.value()
        arr.forEach(v => resultSet.add(v))

        // 2.取出另一个集合的值
        arr = otherSet.value()
        arr.forEach(v => resultSet.add(v))

        return resultSet
    }

    // 交集
    intersection(otherSet) {
        let resultSet = new Set()
        let arr = this.value()

        arr.forEach(v => {
            if(otherSet.has(v)) {
                resultSet.add(v)
            }
        })
        return resultSet
    }

    // 差集
    difference(otherSet) {
        let resultSet = new Set()
        let arr = this.value()

        arr.forEach(v => {
            if(!otherSet.has(v)) {
                resultSet.add(v)
            }
        })
    }
}
```
